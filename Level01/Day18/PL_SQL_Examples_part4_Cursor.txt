/* =========================================================
   4. SQL%FOUND Example

   SQL%FOUND returns TRUE if DML affects at least one row.
   ========================================================= */

BEGIN
    UPDATE employees
    SET salary = salary + 50000
    WHERE employee_id = 9;

    IF SQL%FOUND THEN
        DBMS_OUTPUT.PUT_LINE('SQL%FOUND: Employee salary updated.');
    END IF;

    COMMIT;
END;
/

/* =========================================================
   5. SQL%NOTFOUND Example

   SQL%NOTFOUND returns TRUE if DML affects no row.
   ========================================================= */

BEGIN
    UPDATE employees
    SET salary = salary + 50000
    WHERE employee_id = 999;

    IF SQL%NOTFOUND THEN
        DBMS_OUTPUT.PUT_LINE('SQL%NOTFOUND: No employee found.');
    END IF;

    COMMIT;
END;
/

/* =========================================================
   6. SQL%ROWCOUNT Example

   SQL%ROWCOUNT returns number of affected rows.
   ========================================================= */

BEGIN
    UPDATE employees
    SET salary = salary + 100000
    WHERE department = 'IT';

    DBMS_OUTPUT.PUT_LINE('SQL%ROWCOUNT: Updated rows = ' || SQL%ROWCOUNT);

    COMMIT;
END;

/* =========================================================
   7. SQL%ISOPEN Example

   For implicit cursor, SQL%ISOPEN is always FALSE.
   Oracle automatically opens and closes implicit cursor.
   ========================================================= */

BEGIN
    UPDATE employees
    SET salary = salary
    WHERE department = 'HR';

    IF SQL%ISOPEN THEN
        DBMS_OUTPUT.PUT_LINE('Implicit cursor is open.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('SQL%ISOPEN: Implicit cursor is always closed.');
    END IF;

    COMMIT;
END;
/

   ========================================================= */

/* =========================================================
   8. Explicit Cursor Example
   ========================================================= */

DECLARE
    CURSOR cur_emp IS
        SELECT employee_id, employee_name, department, salary
        FROM employees
        WHERE status = 'ACTIVE';
    v_employee_id   employees.employee_id%TYPE;
    v_employee_name employees.employee_name%TYPE;
    v_department    employees.department%TYPE;
    v_salary        employees.salary%TYPE;
BEGIN
    OPEN cur_emp;

    LOOP
        FETCH cur_emp
        INTO v_employee_id, v_employee_name, v_department, v_salary;
        EXIT WHEN cur_emp%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE(
            v_employee_id || ' - ' ||
            v_employee_name || ' - ' ||
            v_department || ' - ' ||
            v_salary
        );
    END LOOP;
    CLOSE cur_emp;
END;
/


SELECT employee_id, employee_name, department, salary
        FROM employees
        WHERE status = 'ACTIVE';

/* =========================================================
   9. Explicit Cursor with Cursor Attributes
   ========================================================= */

DECLARE
    CURSOR cur_customer IS
        SELECT customer_id, customer_name, balance
        FROM customers
        WHERE status = 'ACTIVE';

    v_customer_id   customers.customer_id%TYPE;
    v_customer_name customers.customer_name%TYPE;
    v_balance       customers.balance%TYPE;
BEGIN
    OPEN cur_customer;

    LOOP
        FETCH cur_customer
        INTO v_customer_id, v_customer_name, v_balance;

        EXIT WHEN cur_customer%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE(
            'Customer: ' || v_customer_name ||
            ', Balance: ' || v_balance
        );

        DBMS_OUTPUT.PUT_LINE(
            'Fetched Row Count: ' || cur_customer%ROWCOUNT
        );
    END LOOP;

    IF cur_customer%ISOPEN THEN
        DBMS_OUTPUT.PUT_LINE('Cursor is still open.');
    END IF;

    CLOSE cur_customer;
   
END;
/

/* =========================================================
   PART C: CURSOR FOR LOOP
   Cursor FOR LOOP automatically:
   - Opens cursor
   - Fetches rows
   - Closes cursor
   ========================================================= */

BEGIN
    FOR emp_rec IN
    (
        SELECT employee_id, employee_name, department, salary
        FROM employees
        WHERE status = 'ACTIVE'
    )
    LOOP
        DBMS_OUTPUT.PUT_LINE(
            emp_rec.employee_id || ' - ' ||
            emp_rec.employee_name || ' - ' ||
            emp_rec.department || ' - ' ||
            emp_rec.salary
        );
    END LOOP;
END;
/

/* =========================================================
   PART D: PARAMETERIZED CURSOR
   Parameterized cursor accepts input values.
   ========================================================= */

DECLARE
    CURSOR cur_emp_by_dept
    (
        p_department employees.department%TYPE
    )
    IS
        SELECT employee_id, employee_name, salary
        FROM employees
        WHERE department = p_department
        AND status = 'ACTIVE';

BEGIN
    DBMS_OUTPUT.PUT_LINE('Employees from IT Department');

    FOR emp_rec IN cur_emp_by_dept('IT')
    LOOP
        DBMS_OUTPUT.PUT_LINE(
            emp_rec.employee_id || ' - ' ||
            emp_rec.employee_name || ' - ' ||
            emp_rec.salary
        );
    END LOOP;

    DBMS_OUTPUT.PUT_LINE('Employees from Finance Department');

    FOR emp_rec IN cur_emp_by_dept('Finance')
    LOOP
        DBMS_OUTPUT.PUT_LINE(
            emp_rec.employee_id || ' - ' ||
            emp_rec.employee_name || ' - ' ||
            emp_rec.salary
        );
    END LOOP;
END;
/

/* =========================================================
   LAB 1: EMPLOYEE SALARY REPORT
   Requirement:
   - Show active employees
   - Calculate annual salary
   - Classify salary level
   ========================================================= */

DECLARE
    CURSOR cur_salary_report IS
        SELECT employee_id, employee_name, department, salary
        FROM employees
        WHERE status = 'ACTIVE'
        ORDER BY department, employee_id;

    v_annual_salary NUMBER;
    v_salary_level  VARCHAR2(30);
BEGIN
    DBMS_OUTPUT.PUT_LINE('===== EMPLOYEE SALARY REPORT =====');

    FOR emp_rec IN cur_salary_report
    LOOP
        v_annual_salary := emp_rec.salary * 12;

        IF emp_rec.salary >= 1000000 THEN
            v_salary_level := 'HIGH SALARY';
        ELSIF emp_rec.salary >= 700000 THEN
            v_salary_level := 'MEDIUM SALARY';
        ELSE
            v_salary_level := 'LOW SALARY';
        END IF;

        DBMS_OUTPUT.PUT_LINE(
            'ID: ' || emp_rec.employee_id ||
            ', Name: ' || emp_rec.employee_name ||
            ', Dept: ' || emp_rec.department ||
            ', Monthly Salary: ' || emp_rec.salary ||
            ', Annual Salary: ' || v_annual_salary ||
            ', Level: ' || v_salary_level
        );
    END LOOP;
END;
/

/* =========================================================
   LAB 2: CUSTOMER STATEMENT GENERATOR
   Requirement:
   - Generate statement for active customers
   - Show customer balance
   - Classify customer account status
   ========================================================= */

DECLARE
    CURSOR cur_customer_statement IS
        SELECT customer_id, customer_name, city, balance
        FROM customers
        WHERE status = 'ACTIVE'
        ORDER BY customer_id;

    v_balance_status VARCHAR2(50);
BEGIN
    DBMS_OUTPUT.PUT_LINE('===== CUSTOMER STATEMENT =====');

    FOR cust_rec IN cur_customer_statement
    LOOP
        IF cust_rec.balance >= 200000 THEN
            v_balance_status := 'GOOD BALANCE';
        ELSIF cust_rec.balance >= 100000 THEN
            v_balance_status := 'NORMAL BALANCE';
        ELSE
            v_balance_status := 'LOW BALANCE';
        END IF;

        DBMS_OUTPUT.PUT_LINE(
            'Customer ID: ' || cust_rec.customer_id ||
            ', Name: ' || cust_rec.customer_name ||
            ', City: ' || cust_rec.city ||
            ', Balance: ' || cust_rec.balance ||
            ', Status: ' || v_balance_status
        );
    END LOOP;
END;
/

/* =========================================================
   LAB 3: TELECOM SUBSCRIBER PROCESSING
   Requirement:
   - Process active telecom subscribers
   - Check prepaid/postpaid plan
   - Warn prepaid users with low balance
   ========================================================= */

DECLARE
    CURSOR cur_subscriber
    (
        p_status telecom_subscribers.status%TYPE
    )
    IS
        SELECT subscriber_id, msisdn, customer_name, plan_name, balance
        FROM telecom_subscribers
        WHERE status = p_status
        ORDER BY subscriber_id;

    v_message VARCHAR2(200);
BEGIN
    DBMS_OUTPUT.PUT_LINE('===== TELECOM SUBSCRIBER PROCESSING =====');

    FOR sub_rec IN cur_subscriber('ACTIVE')
    LOOP
        IF sub_rec.plan_name = 'PREPAID' AND sub_rec.balance <= 2000 THEN
            v_message := 'LOW BALANCE WARNING';
        ELSIF sub_rec.plan_name = 'PREPAID' THEN
            v_message := 'PREPAID ACCOUNT NORMAL';
        ELSIF sub_rec.plan_name = 'POSTPAID' THEN
            v_message := 'POSTPAID BILLING ACCOUNT';
        ELSE
            v_message := 'UNKNOWN PLAN';
        END IF;

        DBMS_OUTPUT.PUT_LINE(
            'Subscriber ID: ' || sub_rec.subscriber_id ||
            ', MSISDN: ' || sub_rec.msisdn ||
            ', Name: ' || sub_rec.customer_name ||
            ', Plan: ' || sub_rec.plan_name ||
            ', Balance: ' || sub_rec.balance ||
            ', Message: ' || v_message
        );
    END LOOP;
END;
/
